## 语法欠缺
- if else if
```cpp
auto check = [&nums, &n, &target](int mid) -> bool {
	bool ans = false;
	if (nums[mid] > nums[n-1])
		if (target > nums[n-1] && target < nums[mid]) ans = true;
	// 这里的else 匹配最接近的if，而不是最初的if
	else if (target > nums[n-1] || nums[mid] > target) ans = true;
	
	return ans;
};

auto check = [&nums, &n, &target](int mid) -> bool {
	bool ans = false;
	if (nums[mid] > nums[n-1])
	{
		if (target > nums[n-1] && target < nums[mid]) ans = true;
	}
	else if (target > nums[n-1] || nums[mid] > target) ans = true;
	
	return ans;
};
```
